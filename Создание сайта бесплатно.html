html
<!DOCTYPE html>
<html lang="ru">
<head>
  <meta charset="UTF-8" />
  <meta name="viewport" content="width=device-width, initial-scale=1.0"/>
  <title>Мой магазин</title>
  <link rel="stylesheet" href="style.css" />
</head>
<body>
  <header>
    <h1>Мой Магазин</h1>
    <nav>
      <a href="#">Главная</a>
      <a href="#">Каталог</a>
      <a href="#">Контакты</a>
    </nav>
  </header>

  <main class="products">
    <div class="product-card">
      <img src="https://via.placeholder.com/150" alt="Товар 1">
      <h3>Товар 1</h3>
      <p>Цена: 1000₽</p>
      <button>Купить</button>
    </div>
    <div class="product-card">
      <img src="https://via.placeholder.com/150" alt="Товар 2">
      <h3>Товар 2</h3>
      <p>Цена: 1500₽</p>
      <button>Купить</button>
    </div>
    <!-- Добавь больше товаров по желанию -->
  </main>

  <footer>
    <p>&copy; 2025 Мой Магазин | Контакты: info@example.com</p>
  </footer>
</body>
</html>
